Максимальное число в массиве

Создадим новое консольное приложение C#

const int N = 10; // размер массива
int[] array = new int[N]; // создадим массив размером N
Random rand = new Random();
// инициализируем массив случайным числами до 100
for( int i = 0; i < array.Length; i++) {
    array[i] = rand.Next(100);
}
// выведем массив на экран
for( int i = 0; i < array.Length; i++) {
    Console.Write($"{array[i]} ");
}
Console.WriteLine();


// найдем максимальное число в массиве
int max = array[0]; // в начале полагаем, что первое число в массиве является максимальным. копируем его в переменнную (аккумулятор) max
for ( int i = 0; i < N; i++) {
    if( array[i] > max) {
        max = array[i];
    }
}

Console.WriteLine($"Максимальное число в массиве = {max}");
